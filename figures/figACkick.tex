\begin{tikzpicture}[x=1.5cm, y=1.5cm]
    \filldraw[thick,pattern=north west lines, pattern color=gray] (-0.5, -1) rectangle (1,1);
    \node[above] at (0.25, -1.0) {\small AC-dipole};
    \draw[thick, red] (-3,0) -- (-0.5,0);
    \foreach \radius in {8, 4, 2.9, 2.1}
    {
        \pgfmathsetmacro\stop{-acos(1.5/\radius)}
        \draw[thick, red] (-.5,0) arc (-90:\stop:\radius);
        \pgfmathsetmacro\yend{\radius*(sin(\stop)+1)}
        \pgfmathsetmacro\ynext{\yend+0.5*tan(\stop+90)}
        \draw[->, thick,red] (1,\yend) -- (1.5,\ynext);
    }
\end{tikzpicture}
